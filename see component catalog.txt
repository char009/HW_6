#include <FL/Fl.H>
#include <FL/Fl_Window.H>
#include <FL/Fl_Text_Display.H>
#include <string>
#include <string.h>
#include <stdio.h>
#include <iostream>
#include <fstream>
#include <FL/Fl_Input.H>
#include <FL/Fl_Return_Button.H>

Fl_Input *choice;
Fl_Window *win = new Fl_Window(640, 400);
using namespace std;


int main() {
	
	Fl_Text_Buffer *buff = new Fl_Text_Buffer();
	Fl_Text_Display *disp = new Fl_Text_Display(20, 20, 640 - 20, 400 - 20, "Component Catalog");
	disp->buffer(buff);
	win->resizable(*disp);
	//win->show();
	string line, display_line="";
	ifstream myfile("catalog.txt");
	int i = 1;
	if (myfile.is_open())
	{
		while (getline(myfile, line))
		{	
			line = std::to_string(i)+". " + line;
			line += "\n";
			cout << line;
			display_line+=line;
			//cout << i<<". "<< display_line;
			i++;
			}
		myfile.close();
	}

	else cout << "Unable to open file";
	
	const char *line1 = display_line.c_str();
	buff->text(line1);

	win->show();
	return(Fl::run());
}
